GameOfLife-OpenGL

This is the OpenGL abstraction part of this project. It is also intended for use in other projects such as Amber. It's released on JitPack.

pom.xml

<repositories>
    <repository>
        <id>jitpack.io</id>
        <url>https://jitpack.io</url>
    </repository>
</repositories>

<dependencies>
    <dependency>
        <groupId>com.github.Redstonecrafter0.GameOfLife</groupId>
        <artifactId>opengl</artifactId>
        <version>{opengl.version}</version>
    </dependency>
</dependencies>
